model = genai.GenerativeModel('gemini-1.5-flash')
chat_histories = {}

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text("Hey! I'm your AI chat bot 🤖\nAsk me anything and I'll reply.")

async def reset(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = update.message.from_user.id
    if user_id in chat_histories:
        del chat_histories[user_id]
    await update.message.reply_text("Chat history cleared! We can start fresh.")

async def chat(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = update.message.from_user.id
    user_message = update.message.text
    await update.message.chat.send_action("typing")
    
    if user_id not in chat_histories:
        chat_histories[user_id] = model.start_chat(history=[])
    
    try:
        response = chat_histories[user_id].send_message(user_message)
        await update.message.reply_text(response.text)
    except Exception as e:
        await update.message.reply_text("Oops, something went wrong. Try again.")
